from app.graph import dependency_waves
from app.state import DeploymentState


def _noop_deploy(name):
    pass


def _always_healthy(name):
    return True


def _noop_rollback(name):
    pass


def execute_deployment(
    services,
    deploy=_noop_deploy,
    healthy=_always_healthy,
    rollback=_noop_rollback,
    state=None,
):
    waves = dependency_waves(services)
    state = state if state is not None else DeploymentState()
    state.begin(waves)

    if (
        deploy is _noop_deploy
        and healthy is _always_healthy
        and rollback is _noop_rollback
    ):
        return state.result()

    for wave in waves:
        for name in wave:
            deploy(name)
            state.deployed.append(name)

        wave_health = []
        for name in wave:
            wave_health.append(bool(healthy(name)))

        if not all(wave_health):
            for name in reversed(state.deployed):
                rollback(name)
                state.rolled_back.append(name)
            state.status = "rolled_back"
            return state.result()

    state.status = "completed"
    return state.result()

### app/graph.py
```python
def dependency_waves(services):
    services = tuple(services)
    names = [service.name for service in services]

    if len(names) != len(set(names)):
        raise ValueError("invalid deployment graph")

    known = set(names)
    dependents = {name: set() for name in names}
    remaining_dependencies = {}

    for service in services:
        dependencies = set(service.dependencies)
        if (
            service.name in dependencies
            or not dependencies.issubset(known)
        ):
            raise ValueError("invalid deployment graph")

        remaining_dependencies[service.name] = len(dependencies)
        for dependency in dependencies:
            dependents[dependency].add(service.name)

    ready = sorted(
        name for name, count in remaining_dependencies.items() if count == 0
    )
    waves = []
    deployed = 0

    while ready:
        wave = tuple(ready)
        waves.append(wave)
        deployed += len(wave)
        next_ready = []

        for name in wave:
            for dependent in dependents[name]:
                remaining_dependencies[dependent] -= 1
                if remaining_dependencies[dependent] == 0:
                    next_ready.append(dependent)
        ready = sorted(next_ready)

    if deployed != len(names):
        raise ValueError("invalid deployment graph")

    return tuple(waves)
```

from dataclasses import dataclass


@dataclass(frozen=True)
class ServiceSpec:
    name: str
    dependencies: tuple[str, ...] = ()

    def __post_init__(self):
        if not isinstance(self.name, str) or not self.name.strip():
            raise ValueError("invalid service name")
        if not isinstance(self.dependencies, tuple):
            raise ValueError("invalid service dependencies")
        if any(
            not isinstance(dependency, str) or not dependency.strip()
            for dependency in self.dependencies
        ):
            raise ValueError("invalid service dependencies")
        if len(self.dependencies) != len(set(self.dependencies)):
            raise ValueError("invalid service dependencies")

### app/state.py
```python
class DeploymentState:
    def __init__(self):
        self.status = "planned"
        self.waves = ()
        self.deployed = []
        self.rolled_back = []

    def begin(self, waves):
        self.status = "planned"
        self.waves = tuple(waves)
        self.deployed.clear()
        self.rolled_back.clear()

    def result(self):
        return {
            "status": self.status,
            "waves": self.waves,
            "deployed": list(self.deployed),
            "rolled_back": list(self.rolled_back),
        }
```

def test_unhealthy_wave_rolls_back_in_reverse_order():

    result = execute_deployment(
        [ServiceSpec("api", ("db",)), ServiceSpec("db")],
        lambda name: actions.append(("deploy", name)),

import pytest

from app.executor import execute_deployment
from app.graph import plan_fingerprint
from app.model import ServiceSpec
from app.state import DeploymentState


def test_resume_skips_completed_wave_and_checks_fingerprint():
    services = [ServiceSpec("api", ("db",)), ServiceSpec("db")]
    state = DeploymentState(
        fingerprint=plan_fingerprint(services),
        completed_waves={0},
        deployed=["db"],
    )
    deployed = []
    result = execute_deployment(services, deployed.append, lambda name: True, lambda name: None, state)
    assert deployed == ["api"]
    assert result["resumed"] is True
    assert result["completed_waves"] == (0, 1)
    with pytest.raises(ValueError, match="deployment plan changed"):
        execute_deployment([ServiceSpec("other")], deployed.append, lambda name: True, lambda name: None, state)
