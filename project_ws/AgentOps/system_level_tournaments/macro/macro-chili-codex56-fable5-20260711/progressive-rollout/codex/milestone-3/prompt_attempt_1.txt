```python
import hashlib


def stable_bucket(subject_id: str, salt: str) -> int:
    if not isinstance(subject_id, str) or not subject_id.strip():
        raise ValueError("subject_id and salt must be non-empty")
    if not isinstance(salt, str) or not salt.strip():
        raise ValueError("subject_id and salt must be non-empty")

    digest = hashlib.sha256(f"{salt}:{subject_id}".encode("utf-8")).digest()
    return int.from_bytes(digest[:8], byteorder="big") % 100
```

```python
from dataclasses import dataclass
from typing import FrozenSet, Mapping


@dataclass(frozen=True)
class RolloutConfig:
    enabled: bool = False
    percentage: int = 0
    salt: str = ""
    allowlist: FrozenSet[str] = frozenset()
    denylist: FrozenSet[str] = frozenset()

    @classmethod
    def from_mapping(cls, values: Mapping[str, str]) -> "RolloutConfig":
        try:
            enabled_value = values["ENABLED"]
            if not isinstance(enabled_value, str):
                raise ValueError

            normalized_enabled = enabled_value.strip().lower()
            if normalized_enabled == "true":
                enabled = True
            elif normalized_enabled == "false":
                enabled = False
            else:
                raise ValueError

            percentage_value = values.get("PERCENTAGE", "0")
            if not isinstance(percentage_value, str):
                raise ValueError
            percentage = int(percentage_value)
            if not 0 <= percentage <= 100:
                raise ValueError

            salt = values["SALT"]
            if not isinstance(salt, str) or not salt.strip():
                raise ValueError

            allowlist = cls._parse_list(values.get("ALLOWLIST", ""))
            denylist = cls._parse_list(values.get("DENYLIST", ""))
            if allowlist & denylist:
                raise ValueError
        except (KeyError, TypeError, ValueError):
            raise ValueError("invalid rollout config") from None

        return cls(
            enabled=enabled,
            percentage=percentage,
            salt=salt,
            allowlist=allowlist,
            denylist=denylist,
        )

    @staticmethod
    def _parse_list(value: str) -> FrozenSet[str]:
        if not isinstance(value, str):
            raise ValueError
        return frozenset(item.strip() for item in value.split(",") if item.strip())
```

### app/decision.py

```python
from typing import Any

from app.cohort import stable_bucket
from app.config import RolloutConfig


def decide_rollout(subject_id: str, config: RolloutConfig) -> dict[str, Any]:
    if subject_id in config.denylist:
        return {"enabled": False, "reason": "denylist", "bucket": None}

    if not config.enabled:
        return {"enabled": False, "reason": "disabled", "bucket": None}

    if subject_id in config.allowlist:
        return {
            "enabled": True,
            "reason": "allowlist",
            "bucket": None,
        }

    bucket = stable_bucket(subject_id, config.salt)
    if bucket < config.percentage:
        return {"enabled": True, "reason": "cohort", "bucket": bucket}

    return {
        "enabled": False,
        "reason": "outside_cohort",
        "bucket": bucket,
    }
```

### app/service.py
```python
from typing import Any, Mapping

from app.config import RolloutConfig
from app.decision import decide_rollout


def evaluate_request(
    subject_id: str,
    raw_config: Mapping[str, str],
    audit: list[dict[str, Any]] | None = None,
) -> dict[str, Any]:
    config = RolloutConfig.from_mapping(raw_config)
    decision = decide_rollout(subject_id, config)
    if audit is not None:
        audit.append({"subject_id": subject_id, **decision})
    return decision
```
