### app/cohort.py
```python
import hashlib


def stable_bucket(subject_id: str, salt: str) -> int:
    if not isinstance(subject_id, str) or not subject_id.strip():
        raise ValueError("subject_id must be a non-empty string")
    if not isinstance(salt, str) or not salt.strip():
        raise ValueError("salt must be a non-empty string")
    digest = hashlib.sha256(f"{salt}:{subject_id}".encode("utf-8")).digest()
    return int.from_bytes(digest[:8], "big") % 100
```

### app/config.py
```python
from dataclasses import dataclass
from typing import Any, Mapping

_INVALID_CONFIG = "invalid rollout config"
_TRUE_VALUES = frozenset({"true", "1"})
_FALSE_VALUES = frozenset({"false", "0"})


def _parse_enabled(value: Any) -> bool:
    if isinstance(value, bool):
        return value
    if isinstance(value, str):
        lowered = value.strip().lower()
        if lowered in _TRUE_VALUES:
            return True
        if lowered in _FALSE_VALUES:
            return False
    raise ValueError(_INVALID_CONFIG)


def _parse_percentage(value: Any) -> int:
    if isinstance(value, bool):
        raise ValueError(_INVALID_CONFIG)
    if isinstance(value, int):
        percentage = value
    elif isinstance(value, str):
        try:
            percentage = int(value.strip())
        except ValueError:
            raise ValueError(_INVALID_CONFIG) from None
    else:
        raise ValueError(_INVALID_CONFIG)
    if not 0 <= percentage <= 100:
        raise ValueError(_INVALID_CONFIG)
    return percentage


def _parse_salt(value: Any) -> str:
    if not isinstance(value, str) or not value.strip():
        raise ValueError(_INVALID_CONFIG)
    return value


@dataclass(frozen=True)
class RolloutConfig:
    enabled: bool = False
    percentage: int = 0
    salt: str = ""

    @classmethod
    def from_mapping(cls, values: Mapping[str, str]) -> "RolloutConfig":
        return cls(
            enabled=_parse_enabled(values.get("ENABLED")),
            percentage=_parse_percentage(values.get("PERCENTAGE")),
            salt=_parse_salt(values.get("SALT")),
        )
```

### app/decision.py
```python
from typing import Any

from app.cohort import stable_bucket
from app.config import RolloutConfig


def decide_rollout(subject_id: str, config: RolloutConfig) -> dict[str, Any]:
    if not config.enabled:
        return {"enabled": False, "reason": "disabled", "bucket": None}
    bucket = stable_bucket(subject_id, config.salt)
    if bucket < config.percentage:
        return {"enabled": True, "reason": "cohort", "bucket": bucket}
    return {"enabled": False, "reason": "outside_cohort", "bucket": bucket}
```
