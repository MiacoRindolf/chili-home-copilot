```python
class CacheStore:
    def __init__(self):
        self._entries = {}
        self._latest = {}

    def get(self, token):
        return self._entries.get(token.entity_id, {}).get(token.revision)

    def put(self, token, value):
        latest = self._latest.get(token.entity_id)
        if latest is not None and token.revision < latest:
            return False
        self._entries.setdefault(token.entity_id, {})[token.revision] = value
        if latest is None or token.revision > latest:
            self._latest[token.entity_id] = token.revision
        return True

    def invalidate_before(self, token):
        revisions = self._entries.get(token.entity_id, {})
        stale = [revision for revision in revisions if revision < token.revision]
        for revision in stale:
            del revisions[revision]
        return len(stale)
```

```python
from dataclasses import dataclass


@dataclass(frozen=True)
class RevisionToken:
    entity_id: str
    revision: int

    @classmethod
    def from_payload(cls, value):
        if not isinstance(value, dict):
            raise ValueError("invalid revision token")
        entity_id = str(value.get("entity_id", "")).strip()
        try:
            revision = int(value.get("revision", 0))
        except (TypeError, ValueError):
            raise ValueError("invalid revision token")
        if not entity_id or revision < 1:
            raise ValueError("invalid revision token")
        return cls(entity_id, revision)
```

### service/catalog.py
```python
from domain.revision import RevisionToken


def load_catalog(token_payload, store, fetcher):
    token = RevisionToken.from_payload(token_payload)
    cached = store.get(token)
    if cached is not None:
        return {"source": "cache", "value": cached, "revision": token.revision}
    value = fetcher(token.entity_id, token.revision)
    if store.put(token, value):
        return {"source": "origin", "value": value, "revision": token.revision}
    return {"source": "stale_rejected", "value": None, "revision": token.revision}
```
